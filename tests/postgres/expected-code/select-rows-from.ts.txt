import type pg from 'pg';

export type SelectRowsFromResult = {
	a: int | null;
	b: int | null;
}

export async function selectRowsFrom(client: pg.Client | pg.Pool | pg.PoolClient): Promise<SelectRowsFromResult[]> {
	const sql = `
	SELECT * FROM ROWS FROM (generate_series(1,3), generate_series(10,12)) AS t(a,b)
	`
	return client.query({ text: sql, rowMode: 'array' })
		.then(res => res.rows.map(row => mapArrayToSelectRowsFromResult(row)));
}

function mapArrayToSelectRowsFromResult(data: any) {
	const result: SelectRowsFromResult = {
		a: data[0],
		b: data[1]
	}
	return result;
}
