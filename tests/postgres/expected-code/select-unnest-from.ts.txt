import type pg from 'pg';

export type SelectUnnestFromResult = {
	id: number | null;
}

export async function selectUnnestFrom(client: pg.Client | pg.Pool | pg.PoolClient): Promise<SelectUnnestFromResult[]> {
	const sql = `
	SELECT * FROM unnest(ARRAY[1, 2, 3]) AS t(id)
	`
	return client.query({ text: sql, rowMode: 'array' })
		.then(res => res.rows.map(row => mapArrayToSelectUnnestFromResult(row)));
}

function mapArrayToSelectUnnestFromResult(data: any) {
	const result: SelectUnnestFromResult = {
		id: data[0]
	}
	return result;
}
